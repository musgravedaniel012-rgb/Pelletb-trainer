        function launchPracticeSession(qs, title, seconds) {
            currentQuestions = qs;
            currentIndex = 0;

        function getRandomQuestions(count, categories) {
            let filtered = questionBank.filter(q => categories.includes(q.category));
            // shuffle
            filtered.sort(() => Math.random() - 0.5);
            return filtered.slice(0, count);

        function renderCurrentQuestion() {
            const q = currentQuestions[currentIndex];
            const area = document.getElementById('question-area');
            let html = `<div style="margin-bottom:16px;"><span class="category-tag">${q.category.toUpperCase()}</span></div>`;
            
            if (q.passage) {
                html += `<div class="reading-passage">${q.passage}</div>`;
            }
            
            html += `<div style="font-size:18px;line-height:1.6;margin-bottom:24px;">${q.q}</div>`;
            html += `<div class="options">`;
            
            q.options.forEach((opt, i) => {
                html += `<div class="option" onclick="selectAnswer(${i})">${String.fromCharCode(65+i)}. ${opt}</div>`;
            });
            
            html += `</div>`;
            area.innerHTML = html;
            document.getElementById('feedback').style.display = 'none';

        let selectedAns = -1;
        
        function selectAnswer(idx) {
            selectedAns = idx;
            const options = document.querySelectorAll('.option');
            options.forEach((o,i) => {
                if (i === idx) o.classList.add('selected');
            });
            
            const q = currentQuestions[currentIndex];
            const isCorrect = idx === q.correct;
            
            if (isCorrect) score++;
            
            // record for progress
            progress.totalAnswered++;
            const catAcc = progress.categoryAccuracy[q.category] || 50;
            progress.categoryAccuracy[q.category] = Math.round((catAcc * 0.9) + (isCorrect ? 10 : 0));
            
            // feedback
            const fb = document.getElementById('feedback');
            fb.innerHTML = `
                <div style="padding:16px;background:${isCorrect?'#004d26':'#4d1a1a'};border-radius:12px;color:#fff;">
                    ${isCorrect ? '✅ Correct!' : '❌ Incorrect'}<br>
                    <strong>Explanation:</strong> ${q.exp}
                </div>
            `;
            fb.style.display = 'block';
            
            // next after 1.8s
            setTimeout(() => {
                currentIndex++;
                if (currentIndex < currentQuestions.length) {
                    selectedAns = -1;
                    renderCurrentQuestion();
                } else {
                    finishSession();
                }
            }, 1800);
        }

        function startTimer(seconds, onFinish) {
            if (timerInterval) clearInterval(timerInterval);
            timeLeft = seconds;
            timerInterval = setInterval(() => {
                timeLeft--;
                // could show timer if wanted, but for drill it's background
                if (timeLeft <= 0) {
                    clearInterval(timerInterval);
                    onFinish();
                }
            }, 1000);
        }

        function finishSession() {
            if (timerInterval) clearInterval(timerInterval);
            
            const percent = Math.round((score / currentQuestions.length) * 100);
            progress.accuracy = Math.round((progress.accuracy * 0.85) + (percent * 0.15));
            progress.tScore = Math.max(30, Math.min(80, Math.round(38 + (percent * 0.45))));
            
            saveProgress();
            updateHeaderScore();
            updateDashboardStats();
            
            // show result modal
            const modal = document.getElementById('result-modal');
            document.getElementById('modal-title').textContent = percent > 80 ? "Excellent Work!" : "Session Complete";
            document.getElementById('modal-score').textContent = percent + "%";
            document.getElementById('modal-tscore').innerHTML = `Estimated T-Score: <strong>${progress.tScore}</strong>`;
            document.getElementById('modal-detail').innerHTML = `
                Correct: ${score} / ${currentQuestions.length}<br>
                Category accuracy updated
            `;
            modal.classList.add('active');
        }
